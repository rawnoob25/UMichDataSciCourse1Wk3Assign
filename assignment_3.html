def answer_one():
    import pandas as pd
    import re
    col_names = ['Country', 'Energy Supply', 'Energy Supply per Capita', '% Renewable']
    energy = pd.read_excel('Energy Indicators.xls',skiprows = 17, skip_footer = 38, parse_cols = "C:G", names = col_names, na_values = '...')
    energy['Energy Supply']=10**6*energy['Energy Supply']
    energy['Country'].replace(r'[\d]','', inplace = True, regex = True) #replace  digits with empty space
    energy['Country'].replace(r'\([\S\s]*\)','', inplace = True, regex = True) #any parenthesized text (including the parentheses)
    #with the empty string; also handles nested parentheses (e.g 'Congo (The Democratic Republic (yes it's a republic))is awesome' -> 'Congo is awesome')
    energy_nameChangeMap = {"Republic of Korea": "South Korea", "United States of America": "United States", "United Kingdom of Great Britain and Northern Ireland": "United Kingdom",
               "China, Hong Kong Special Administrative Region": "Hong Kong" }
    
    #----CODE DOWN TO HERE READS 'Energy Indicators.xls', AND MAKES SOME CHANGES TO THE 'Energy Supply' COLUMN AS
    #WELL AS SOME PRELIMINARY CHANGES TO THE 'Country" COLUMN----------------------------------------------------
    
    
    #modify is a function that'll be used to change the Country field of both the energy and the gdp data frames
    def modify(e, d):
        if e in d.keys():
            return d[e]
        else:
            return e.strip()
    energy['Country'] = energy['Country'].apply(modify, args = (energy_nameChangeMap,)) #if function to apply to elements of
    #Pandas series takes more than one argument, pass the other arguments as a tuple bound to parameter name 'args'
    gdp = pd.read_csv('world_bank.csv', header = 4)
    gdp_nameChangeMap = {"Korea, Rep.": "South Korea", "Iran, Islamic Rep.": "Iran", "Hong Kong SAR, China": "Hong Kong"}
    gdp['Country Name'] = gdp['Country Name'].apply(modify, args = (gdp_nameChangeMap,))
    
    
    #----THE ABOVE CHUNK OF CODE READS 'world_bank.csv' AND MAKES CHANGES TO THE 'Country' field in both it
    #and in energy--------------------------------------------------------------------------------------------------
    ScimEn = pd.read_excel('scimagojr-3.xlsx')
    
    gdpColsToKeep = [gdp.columns[0]]+list(gdp.columns[-10:])
    gdpToKeep = gdp.loc[:,gdpColsToKeep]
    gdpToKeep.rename(columns = {'Country Name':'Country'}, inplace = True)
    #ScimEn_tokeep = ScimEn.sort_values(by = 'Rank').query('Rank<=15')
    joined = ScimEn.merge(energy, on = 'Country', how = 'inner').merge(gdpToKeep, on = 'Country', how = 'inner').set_index('Country')
    
    global joinedDf
    joinedDf = copy.deepcopy(joined)
    
    global joinedOuter 
    joinedOuter = ScimEn.merge(energy, on = 'Country', how ='outer').merge(gdpToKeep, on = 'Country', how = 'outer').set_index('Country')
    
    filtered = joined.query('Rank<=15')
    #---THIS FINAL CHUNK OF CODE (ABOVE) READS IN DATA FROM 'scimagojr-3.xlsx' into ScimEn, REMOVES COLUMNS FROM gdp and assigns to gdpToKeep,
    #filters the relevant observations from ScimEn into ScimEn_tokeep, and performs the 3-way merge btwn ScimEn_tokeep,
    # energy and gdpToKeep on the Country field
    return filtered
